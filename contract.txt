type CreateContractForm = {
  // Step 1 (supported today)
  client: string;
  building: string;
  capacity: number | "";
  monthlyRent: number | "";
  contractStartDate: string;   // yyyy-mm-dd
  contractEndDate: string;     // yyyy-mm-dd
  terms: string;
  initialCredits?: number | "";
  creditValueAtSignup?: number | "";

  // Step 2
  billingCycle?: "monthly" | "quarterly" | "annually";
  paymentTerms?: "net_15" | "net_30" | "net_45" | "net_60";
  latePaymentPenalty?: { type: "percentage" | "fixed"; value: number; description?: string };
  priceEscalationClause?: { enabled: boolean; percentage?: number; frequency?: "annual"; effectiveFrom?: string };
  discount?: { type: "percentage" | "fixed"; value: number; reason?: string; validUntil?: string };
  setupFee?: number;
  onboardingFee?: number;
  noticePeriodDays?: number;
  autoRenewal?: boolean;
  renewalNoticePeriod?: number;
  prorateOnStart?: boolean;

  // Step 3
  includedServices?: {
    wifi?: boolean; parking?: boolean; meetingRooms?: boolean; pantry?: boolean;
    reception?: boolean; mailHandling?: boolean; housekeeping?: boolean; printingScanning?: boolean;
  };
  operatingHours?: { weekdays?: string; weekends?: string; publicHolidays?: string };
  accessType?: "24x7" | "business_hours";
  seatingArrangement?: "dedicated_desk" | "hot_desk";
  meetingRoomCredits?: { hoursPerMonth?: number; additionalHourRate?: number };
  printingCredits?: { pagesPerMonth?: number; additionalPageRate?: number };
  parkingSlots?: number;
  lockerFacility?: { included?: boolean; quantity?: number };
  customBranding?: { allowed?: boolean; signagePermissions?: string };

  // Step 4
  authorizedSignatory?: { name?: string; designation?: string; email?: string; phone?: string };
  primaryContact?: { name?: string; designation?: string; email?: string; phone?: string; role?: string };
  billingContact?: { name?: string; email?: string; phone?: string };
  operationsContact?: { name?: string; email?: string; phone?: string };
  complianceDetails?: {
    gstNumber?: string; panNumber?: string; companyRegistrationNumber?: string;
    registeredAddress?: string; stampPaperValue?: number; jurisdiction?: string;
    arbitrationClause?: boolean;
  };

  // Step 5
  attachments?: Array<{ name: string; url?: string; file?: File; type?: string; category?: string }>;
  annexures?: Array<{ title: string; description?: string; fileUrl?: string }>;
  specialClauses?: {
    exclusivity?: boolean; exclusivityDetails?: string | null;
    expansionRights?: boolean; expansionTerms?: string;
    relocationRights?: boolean; relocationTerms?: string;
    subleasing?: boolean; subleasingTerms?: string | null;
    visitorPolicy?: string;
  };
  operational?: {
    moveInDate?: string; handoverDate?: string; fitoutPeriodDays?: number;
    accessCardsIssued?: number; parkingPassesIssued?: number; lockerKeysIssued?: number;
  };
};
