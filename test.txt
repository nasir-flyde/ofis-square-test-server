import React, { useState } from "react";
import { useNavigate } from "react-router-dom";

export default function AuthPage({ onLogin }) {
  const navigate = useNavigate();
  const [authMode, setAuthMode] = useState('password'); // 'password' or 'otp'
  const [formData, setFormData] = useState({
    email: "",
    password: "",
    phone: "",
    otp: ""
  });
  const [errors, setErrors] = useState({});
  const [loading, setLoading] = useState(false);
  const [otpSent, setOtpSent] = useState(false);
  const [countdown, setCountdown] = useState(0);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({ ...prev, [name]: value }));
    if (errors[name]) {
      setErrors(prev => ({ ...prev, [name]: "" }));
    }
  };

  const validateForm = () => {
    const newErrors = {};
    
    if (authMode === 'password') {
      if (!formData.email.trim()) {
        newErrors.email = "Email is required";
      } else if (!/\S+@\S+\.\S+/.test(formData.email)) {
        newErrors.email = "Please enter a valid email";
      }
      
      if (!formData.password.trim()) {
        newErrors.password = "Password is required";
      } else if (formData.password.length < 6) {
        newErrors.password = "Password must be at least 6 characters";
      }
    } else {
      if (!formData.phone.trim()) {
        newErrors.phone = "Phone number is required";
      } else if (!/^[0-9]{10}$/.test(formData.phone.replace(/\D/g, ''))) {
        newErrors.phone = "Please enter a valid 10-digit phone number";
      }
      
      if (otpSent && !formData.otp.trim()) {
        newErrors.otp = "OTP is required";
      }
    }

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSendOtp = async () => {
    if (!formData.phone.trim()) {
      setErrors({ phone: "Phone number is required" });
      return;
    }

    setLoading(true);
    try {
      // Simulate API call
      await new Promise(resolve => setTimeout(resolve, 1000));
      setOtpSent(true);
      setCountdown(60);
      setErrors({});
      
      // Start countdown
      const timer = setInterval(() => {
        setCountdown(prev => {
          if (prev <= 1) {
            clearInterval(timer);
            return 0;
          }
          return prev - 1;
        });
      }, 1000);
    } catch (error) {
      setErrors({ general: "Failed to send OTP. Please try again." });
    } finally {
      setLoading(false);
    }
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    if (!validateForm()) return;
    
    setLoading(true);
    try {
      if (authMode === 'otp' && !otpSent) {
        await handleSendOtp();
        return;
      }
      
      // Simulate login API call
      await new Promise(resolve => setTimeout(resolve, 1500));
      
      // Call the onLogin callback to update authentication state
      if (onLogin) {
        onLogin();
      }
      
      // Navigate to dashboard after successful login
      navigate("/cabins");
    } catch (error) {
      setErrors({ general: "Login failed. Please try again." });
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen flex bg-white dark:bg-darkblack-500">
      {/* Left Panel - Branding */}
      <div className="flex-1 bg-white dark:bg-darkblack-600 flex flex-col justify-center px-16 py-12 relative overflow-hidden">
        {/* Decorative circles */}
        <div className="absolute bottom-0 left-0 w-96 h-96 border-2 border-bgray-200 dark:border-darkblack-400 rounded-full -translate-x-32 translate-y-32"></div>
        <div className="absolute bottom-0 left-0 w-80 h-80 border-2 border-bgray-200 dark:border-darkblack-400 rounded-full -translate-x-24 translate-y-24"></div>
        
        <div className="relative z-10">
          <div className="mb-4">
            <img 
              src="/assets/images/logo/logo-color.svg" 
              alt="Ofis Square" 
              className="h-12 dark:hidden"
            />
            <img 
              src="/assets/images/logo/logo-white.svg" 
              alt="Ofis Square" 
              className="h-12 hidden dark:block"
            />
          </div>
          <p className="text-xl text-bgray-600 dark:text-bgray-300 mb-8">
            Workspaces that work wonders Space
          </p>
          <button className="bg-success-300 hover:bg-success-400 text-white px-6 py-3 rounded-lg font-medium transition-colors duration-200">
            Read More
          </button>
        </div>
      </div>

      {/* Right Panel - Login Form */}
      <div className="flex-1 bg-bgray-50 dark:bg-darkblack-500 flex flex-col justify-center px-16 py-12">
        <div className="max-w-md mx-auto w-full">
          <div className="mb-8">
            <h2 className="text-3xl font-bold text-bgray-900 dark:text-white mb-2">
              Hello Again!
            </h2>
            <p className="text-bgray-600 dark:text-bgray-300">Welcome Back</p>
            
            {/* Auth Mode Toggle */}
            <div className="flex bg-white dark:bg-darkblack-600 rounded-lg p-1 mt-4 border border-bgray-200 dark:border-darkblack-400">
              <button
                type="button"
                onClick={() => {
                  setAuthMode('password');
                  setOtpSent(false);
                  setErrors({});
                  setFormData(prev => ({ ...prev, otp: '', phone: '' }));
                }}
                className={`flex-1 py-2 px-4 rounded-md text-sm font-medium transition-colors duration-200 ${
                  authMode === 'password'
                    ? 'bg-success-300 text-white shadow-sm'
                    : 'text-bgray-600 dark:text-bgray-300 hover:text-bgray-900 dark:hover:text-white'
                }`}
              >
                Password
              </button>
              <button
                type="button"
                onClick={() => {
                  setAuthMode('otp');
                  setErrors({});
                  setFormData(prev => ({ ...prev, email: '', password: '' }));
                }}
                className={`flex-1 py-2 px-4 rounded-md text-sm font-medium transition-colors duration-200 ${
                  authMode === 'otp'
                    ? 'bg-success-300 text-white shadow-sm'
                    : 'text-bgray-600 dark:text-bgray-300 hover:text-bgray-900 dark:hover:text-white'
                }`}
              >
                OTP
              </button>
            </div>
          </div>

          <form onSubmit={handleSubmit} className="space-y-6">
            {errors.general && (
              <div className="bg-error-50 dark:bg-error-300/10 border border-error-200 dark:border-error-300/20 rounded-lg p-4">
                <p className="text-sm text-error-300 dark:text-error-200">{errors.general}</p>
              </div>
            )}

            <div className="space-y-4">
              {authMode === 'password' ? (
                <>
                  <div className="relative">
                    <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none">
                      <svg className="h-5 w-5 text-bgray-400 dark:text-bgray-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M16 12a4 4 0 10-8 0 4 4 0 008 0zm0 0v1.5a2.5 2.5 0 005 0V12a9 9 0 10-9 9m4.5-1.206a8.959 8.959 0 01-4.5 1.207" />
                      </svg>
                    </div>
                    <input
                      type="email"
                      name="email"
                      value={formData.email}
                      onChange={handleChange}
                      placeholder="Email Address"
                      className="w-full pl-12 pr-4 py-4 bg-white dark:bg-darkblack-600 rounded-lg border border-bgray-200 dark:border-darkblack-400 focus:ring-2 focus:ring-success-300 focus:border-transparent focus:outline-none text-bgray-900 dark:text-white placeholder-bgray-500 dark:placeholder-bgray-400 transition-colors duration-200"
                      required
                      autoComplete="email"
                    />
                    {errors.email && <p className="mt-1 text-sm text-error-300">{errors.email}</p>}
                  </div>

                  <div className="relative">
                    <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none">
                      <svg className="h-5 w-5 text-bgray-400 dark:text-bgray-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 15v2m-6 4h12a2 2 0 002-2v-6a2 2 0 00-2-2H6a2 2 0 00-2 2v6a2 2 0 002 2zm10-10V7a4 4 0 00-8 0v4h8z" />
                      </svg>
                    </div>
                    <input
                      type="password"
                      name="password"
                      value={formData.password}
                      onChange={handleChange}
                      placeholder="Password"
                      className="w-full pl-12 pr-4 py-4 bg-white dark:bg-darkblack-600 rounded-lg border border-bgray-200 dark:border-darkblack-400 focus:ring-2 focus:ring-success-300 focus:border-transparent focus:outline-none text-bgray-900 dark:text-white placeholder-bgray-500 dark:placeholder-bgray-400 transition-colors duration-200"
                      required
                      autoComplete="current-password"
                    />
                    {errors.password && <p className="mt-1 text-sm text-error-300">{errors.password}</p>}
                  </div>
                </>
              ) : (
                <>
                  <div className="relative">
                    <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none">
                      <svg className="h-5 w-5 text-bgray-400 dark:text-bgray-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                        <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M3 5a2 2 0 012-2h3.28a1 1 0 01.948.684l1.498 4.493a1 1 0 01-.502 1.21l-2.257 1.13a11.042 11.042 0 005.516 5.516l1.13-2.257a1 1 0 011.21-.502l4.493 1.498a1 1 0 01.684.949V19a2 2 0 01-2 2h-1C9.716 21 3 14.284 3 6V5z" />
                      </svg>
                    </div>
                    <input
                      type="tel"
                      name="phone"
                      value={formData.phone}
                      onChange={handleChange}
                      placeholder="Phone Number"
                      className="w-full pl-12 pr-4 py-4 bg-white dark:bg-darkblack-600 rounded-lg border border-bgray-200 dark:border-darkblack-400 focus:ring-2 focus:ring-success-300 focus:border-transparent focus:outline-none text-bgray-900 dark:text-white placeholder-bgray-500 dark:placeholder-bgray-400 transition-colors duration-200"
                      required
                      autoComplete="tel"
                    />
                    {errors.phone && <p className="mt-1 text-sm text-error-300">{errors.phone}</p>}
                  </div>

                  {otpSent && (
                    <div className="relative">
                      <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none">
                        <svg className="h-5 w-5 text-bgray-400 dark:text-bgray-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z" />
                        </svg>
                      </div>
                      <input
                        type="text"
                        name="otp"
                        value={formData.otp}
                        onChange={handleChange}
                        placeholder="Enter OTP"
                        className="w-full pl-12 pr-4 py-4 bg-white dark:bg-darkblack-600 rounded-lg border border-bgray-200 dark:border-darkblack-400 focus:ring-2 focus:ring-success-300 focus:border-transparent focus:outline-none text-bgray-900 dark:text-white placeholder-bgray-500 dark:placeholder-bgray-400 transition-colors duration-200"
                        required
                        maxLength="6"
                      />
                      {errors.otp && <p className="mt-1 text-sm text-error-300">{errors.otp}</p>}
                      
                      <div className="mt-2 flex justify-between items-center">
                        <p className="text-sm text-bgray-600 dark:text-bgray-400">
                          OTP sent to {formData.phone}
                        </p>
                        {countdown > 0 ? (
                          <p className="text-sm text-bgray-500 dark:text-bgray-400">Resend in {countdown}s</p>
                        ) : (
                          <button
                            type="button"
                            onClick={handleSendOtp}
                            className="text-sm text-success-300 hover:text-success-400 transition-colors duration-200"
                            disabled={loading}
                          >
                            Resend OTP
                          </button>
                        )}
                      </div>
                    </div>
                  )}
                </>
              )}
            </div>

            <button
              type="submit"
              disabled={loading}
              className="w-full bg-success-300 hover:bg-success-400 text-white py-4 rounded-lg font-medium focus:outline-none focus:ring-2 focus:ring-success-300 focus:ring-offset-2 dark:focus:ring-offset-darkblack-500 disabled:opacity-50 disabled:cursor-not-allowed transition-all duration-200"
            >
              {loading ? (
                <div className="flex items-center justify-center">
                  <svg className="animate-spin -ml-1 mr-3 h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                    <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                    <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                  </svg>
                  Processing...
                </div>
              ) : (
                authMode === 'otp' && !otpSent ? "Send OTP" : 
                authMode === 'otp' && otpSent ? "Verify OTP" : 
                "Login"
              )}
            </button>

            <div className="text-center">
              <a
                href="mailto:support@ofissquare.com?subject=Login%20Issue%20-%20Ofis%20Square%20Admin%20Panel"
                className="text-sm text-bgray-600 dark:text-bgray-400 hover:text-bgray-800 dark:hover:text-bgray-200 transition-colors duration-200"
              >
                Forgot Password ?
              </a>
            </div>
          </form>
        </div>
      </div>
    </div>
  );
}
